% Special Equipment Texts

\newcommand{\deathskissdef}{%
	Attacks made with this weapon become \textbf{Magical Attacks} and gain +1 Strength, +1 Armour Penetration, and \textbf{Multiple Wounds (2)}.%
}
\newcommand{\transcendencedef}{%
	Attacks made with this weapon become \textbf{Magical Attacks} and gain \textbf{Lethal Strike}. For \newrule{each} unsaved wound inflicted \newrule{against enemy models} with this weapon, the wielder gains +1 Strength and +1 Armour Penetration for the remainder of the game, to a maximum of +2 each.%
}
\newcommand{\moraecsreapingdef}{%
	The wielder \textbf{always} has at least Attack Value 4. Attacks \newrule{made} with this weapon become \textbf{Magical Attacks}, gain \textbf{Death Trance}, and \textbf{always} have at least Strength 4 and at least Armour Penetration 3. For each unsaved wound inflicted \newrule{against enemy models} with this weapon, the owner gains one Veil Token.%
}

\newcommand{\sealoftherepublicdef}{%
	For each unsaved wound inflicted \newrule{against enemy models} by the wearer's Close Combat Attacks, the wearer gains +1 Armour for the remainder of the game.%
}
\newcommand{\terrifyingvisagerestriction}{Standard Height models only}
\newcommand{\terrifyingvisagedef}{%
	The bearer gains \textbf{Distracting} while using this Shield.%
}

\newcommand{\bannerofgardaecosrestriction}{Cannot be taken by units that count towards Core}
\newcommand{\bannerofgardaecosdef}{%
	All friendly units Engaged in the same Combat as the bearer gain \textbf{Death Trance}.%
}
\newcommand{\bannerofblooddef}{%
	Model parts without Harnessed in the bearer's unit gain \textbf{Devastating Charge (+1 \At{})}.%
}
\newcommand{\academybannerrestriction}{Corsairs, Dread Legionnaires, and Repeater Auxiliaries only}
\newcommand{\academybannerdef}{%
	\rnf{} models in the bearer's unit gain +1 Armour Penetration in the First Round of Combat. Corsairs, Dread Legionnaires, and Repeater Auxiliaries in units within \distance{6} of one or more other units with Academy Banner gain +1 Armour Penetration in the First Round of Combat.%
}

\newcommand{\midnightcloakrestriction}{Cannot be taken by models with Towering Presence}
\newcommand{\midnightcloakdef}{%
	The wearer gains \textbf{Aegis (3+, against Ranged Attacks)} and \textbf{Devastating Charge (Multiple Wounds (D3))}.%
}
\newcommand{\pendantofdisdaindef}{%
	The bearer gains \textbf{Aegis (4+, against Strength 5 or more)}.%
}
\newcommand{\ringofshadowsdef}{%
	If the bearer's unit consists entirely of Standard Height models:%
	\begin{smallitemizeitem}%
		\item All models in the unit gain \textbf{Hard Target (1)}.%
		\item Close Combat Attacks allocated towards models in the bearer's unit suffer \minuss{}1 Offensive Skill.%
	\end{smallitemizeitem}%
}
\newcommand{\beastmasterslashdef}{%
	The roll for determining the number of hits from Stomp Attacks of friendly Gigantic models within \distance{12} of the bearer\newrule{'s model} is subject to Maximised Roll.\newline%
	The roll for determining the number of hits from Stomp Attacks of enemy Gigantic models within \distance{12} of the bearer\newrule{'s model} is subject to Minimised Roll.%
}
\newcommand{\wanderingfamiliardef}{%
	At the start of each friendly Magic Phase, you may place a \newrule{Standard Height model with a 20\timess{}\SI{20}{\milli\meter} base representing the} familiar within \distance{6} of the bearer. It must be placed more than \distance{1} away from \newrule{any} model and from Impassable Terrain. Whenever the bearer casts a non-Bound Spell, it may choose to use the position of the familiar when drawing Line of Sight and measuring Range, and the familiar's Front Arc for Direct spells (it must use all of these or none at all). At the end of the Magic Phase, the familiar is removed.%
}
\newcommand{\amuletofspitedef}{%
	In each friendly Magic Phase, before Siphon the Veil, the owner gains an additional Magic Dice. In each of the opponent's Magic Phases, before Siphon the Veil, the owner must discard a single Magic Dice.%
}
\newcommand{\elixirofshadowsrestriction}{Assassins only}
\newcommand{\elixirofshadowsdef}{%
	One use only. May be activated at the start of any Round of Combat. Until the end of the phase, the bearer gains \textbf{Divine Attacks} and must reroll failed to-wound rolls with its Close Combat Attacks. At the end of the phase, the bearer loses 1 Health Point with no saves of any kind allowed.%
}
