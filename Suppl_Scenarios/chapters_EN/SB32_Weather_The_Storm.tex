
\newnarrativescenario{10}{Weather the Storm}
\label{WeathertheStorm}

\flufffont{To fight a battle in the midst of a blizzard is to lose all sense of what you're actually fighting for. Instead you struggle to simply stay alive.}

\subsection*{Deployment}

\rewordedrule{Standard Deployment Type:} Frontline Clash\rewordedrule{.}

\subsection*{Pre-Game Set-up}

Set up the \rewordedrule{Battlefield} with \rewordedrule{the following Terrain Features:} 1--2 Forest\rewordedrule{s} and 1--3 small Impassable Terrain\rewordedrule{s (approximately \distance{3\timess{}3})}. \rewordedrule{You may use the rules of Building the Battlefield in the Rulebook to randomly place those Terrain Features.}

\subsection*{Scenario Special Rules}

\rewordedrule{At the beginning of each Player Turn, r}oll \rewordedrule{a} D6 and multiply \rewordedrule{the result} by \rewordedrule{6:} \rewordedrule{the result corresponds to the Distance of Sight, indicating} how many inches the \rewordedrule{models} can see into the blizzard. A roll of \result{1} means there is no visibility penalty that \rewordedrule{Player T}urn.

\rewordedrule{A model or a unit} cannot shoot, \rewordedrule{C}harge, or otherwise specifically target a unit \rewordedrule{(including spells)} \rewordedrule{farther away from this model or unit than the} \rewordedrule{D}istance of \rewordedrule{S}ight\rewordedrule{}.

\rewordedrule{In addition, during step 7 of the Deployment Phase Sequence (after moving Vanguarding units but before rolling for first turn)}, each player rolls \rewordedrule{a} D6 and \rewordedrule{the corresponding effects in the table below are immediately} applie\rewordedrule{d to both armies}. \rewordedrule{All effects remain in play until the end of the game unless specifically stated otherwise.} These effects stack \rewordedrule{if both players roll the same result}.

\startweatherthestormtable
\weatherthestormtableentry{1}{Bitter Cold}{%
	All \rewordedrule{models} with an \rewordedrule{A}rmour \rewordedrule{of 2} or \rewordedrule{higher} suffer \minuss{}1 \rewordedrule{Advance Rate and \minuss{}2 March Rate. If a model's Armour is modified to lower than 2, it no longer is affected, and if it is modified to 2 or higher, it will be affected}.%
}
\weatherthestormtableentry{2}{Hopelessly Lost}{%
	\rewordedrule{Each} player must nominate one enemy unit\rewordedrule{, starting with the player that chose their Deployment Zone}. \rewordedrule{Move this unit} \distance{2D6} \rewordedrule{in a random direction (even if the unit normally couldn't move), keeping its orientation, otherwise following the rules for moving a Fleeing unit (like units having to take a Panic Test for friendly units Fleeing through their Unit Boundary).}  \rewordedrule{There are however two exceptions: the unit does not take any Dangerous Terrain Tests, and it follows the rules for Pursuing Off the Board if it comes into contact with the Board Edge. In this case, the unit returns in Game Turn 2 instead of Game Turn 1.}%
}
\weatherthestormtableentry{3}{Frostbite}{%
	\rewordedrule{Roll 3D6. The result is used for both players. They must inflict this amount of Health Point losses among their own army, spreading them across their models however they want.} \rewordedrule{Count the Victory Points for each player after the losses, as if the game was over. Note down the result. These Partial Victory Points will be used when determining who is the winner at the end of the game.}%
}
\weatherthestormtableentry{4}{Broken Morale}{%
	\rewordedrule{Every model} except the \rewordedrule{G}eneral \rewordedrule{suffers \minuss{}1 Discipline}.%
}
\weatherthestormtableentry{5}{Poor Sight}{%
\rewordedrule{The range of Shooting Weapons, Artillery Weapons, and spells with type Missile is reduced} by \distance{6}.%
}
\weatherthestormtableentry{6}{Hazardous Landscape}{%
	All models performing \rewordedrule{a} movement other than a Reform \rewordedrule{during step 2 of the Movement Phase Sequence (Fleeing units and units with Random Movement)} must take a Dangerous Terrain (1) Test.%
}
\closeweatherthestormtable

\subsection*{Winning the Scenario}

Use the standard \rewordedrule{Victory C}onditions of the \rewordedrule{Rulebook}.

\rewordedrule{If the Frostbite effect was rolled at the beginning of the game, count the Victory Points as normal at the end of the game, including the losses from Frostbite, as if those losses happened during the game. The player who scored less Partial Victory Points when the Frostbite effect was applied gains a bonus of Victory Points equal to the difference in Partial Victory Points of both players, to compensate the fact that he effectively played with less Army Points.}
