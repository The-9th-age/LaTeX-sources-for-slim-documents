\flufffont{When two factions both firmly believe they have the ultimate claim to the same land, there can only be war.}

\subsection*{Deployment}

\rewordedrule{Standard Deployment Type:} Frontline Clash\rewordedrule{.}

\subsection*{Pre-Game Set-up}

\rewordedrule{At the end of step 6 of the Pre-Game Sequence (immediately after determining Deployment Zones)}, both players must place a Strategic Objective anywhere on the \rewordedrule{Battlefield} \rewordedrule{entirely} outside their own \rewordedrule{D}eployment \rewordedrule{Z}one \rewordedrule{and at least \distance{1} away from Impassable Terrain, starting with the player who chose their Deployment Zone}. 

\subsection*{Scenario Special Rules}

\newcommand{\strategicobjective}{Strategic Objective}
\newcommand{\unmovable}{Unmoveable}
\newcommand{\unmovabledef}{%
	The Strategic Objective cannot move in any way once deployed\rewordedrule{, and for rules purposes its Pursuit and Overrun Distance is always \distance{0}}. It cannot be joined by Characters.%
}
\newcommand{\themed}{Themed}
\newcommand{\themeddef}{%
	\rewordedrule{During step 7 of the Deployment Phase Sequence (after moving Vanguarding units but before rolling for first turn)}, each player must give their own Strategic Objective 2 \rewordedrule{upgrades} from the \rewordedrule{list} below\rewordedrule{starting with the player who chose their Deployment Zone}:
	\begin{itemize}
	\item \magicresistance{4}
	\item \aegis{5+}
	\item \fortitude{4}
	\item Armour \rewordedrule{\textbf{set} to} 4
	\item \rewordedrule{Toxic Attacks can \textbf{never} wound the model and attacks that are Poison Attacks against the model lose this Attack Attribute}
	\item Resilience \rewordedrule{\textbf{set} to 8}
	\item \distracting{}
	\item \hardtarget{} \rewordedrule{(1)}
	\item Health Points \rewordedrule{\textbf{set} to 9}
	\item At the \rewordedrule{start} of each \rewordedrule{G}ame \rewordedrule{T}urn, roll a D6. On a 5+\rewordedrule{,} this model \rewordedrule{R}ecovers 1 \rewordedrule{Health Point}
	\end{itemize}%
}

\subsection*{Winning the Scenario}

A player is \rewordedrule{immediately} declared the winner \rewordedrule{when} their opponent's Strategic Objective is \rewordedrule{removed as a casualty} or \rewordedrule{when all enemy units have been removed as casualties}. If no winner has been declared after 7 \rewordedrule{Game T}urns, the game is a draw.
