% Special Equipment Texts

\newcommand{\burningportentdef}{%
\enchantmentrestriction{\hw{} enchantment.}
Attacks made with this weapon have their Armour Penetration \textbf{set} to 10, become \textbf{\flamingattacks{}} and \textbf{\magicalattacks{}}, and gain \textbf{\multiplewounds{D3}{}}.%
}
\newcommand{\symbolofslaughterdef}{%
\enchantmentrestriction{\hw{} and \pw{} enchantment.}
When using this weapon, the wielder gains +2 Attack Value, +2 Agility, and \textbf{\magicalattacks{}}. Close Combat Attacks made against the wielder's model gain +1 to hit.%
}

\newcommand{\thriceforgeddef}{%
\itemrestriction{\newrule{Cannot be taken by Gigantic models}.}
\enchantmentrestriction{Suit of Armour enchantment.}
The wearer gains +3 Armour. \newrule{If the bearer's model has Towering Presence, its Armour cannot be improved beyond 5.}%
}
\newcommand{\gladiatorsspiritdef}{%
\enchantmentrestriction{Suit of Armour enchantment.}
The wearer gains +1 Armour and \textbf{Parry}.%
}

\newcommand{\iconoftheinfinitedef}{%
\itemrestriction{Cannot be taken by units that count towards Core. 0--2 per Army.}
The bearer of one or more Icons of the Infinite can cast \spellformat{\hellfire}{\hereditaryspell} as a Bound Spell with Power Level (4/8).%
}
\newcommand{\zealotsbannerdef}{%
\itemrestriction{0--3 per Army.}
The maximum number of Supporting Attacks that models in the second rank of the bearer's unit can perform is increased by 1.%
}
\newcommand{\wastelandtorchdef}{%
The bearer's unit gains \textbf{Strider (Ruins)}. After determining Deployment Zones (at the end of step 6 of the Pre-Game Sequence), you may choose a single Field or Forest Terrain Feature that becomes Ruins. The bearer's unit gains \textbf{\flamingattacks{}} in the First Round of Combat.%
}

\newcommand{\ledgerofsoulsdef}{%
\itemrestriction{Dominant.}
\newrule{Whenever a friendly model without Insignificant in a unit within \distance{9} of the bearer's model loses its last Health Point due to enemy attacks, you gain one Veil Token for each Health Point that model started the game with.}%
}
\newcommand{\veilgateorbdef}{%
\newrule{At the end of Siphon the Veil, the army may keep up to 6 Veil Tokens instead of the normal 3.}%
}
\newcommand{\immortalgauntletsdef}{%
At the start of any Round of Combat you may discard a Veil Token from your Veil Token pool. If so, choose either \divineattacks{}, \flamingattacks{}, or \magicalattacks{}. The bearer's Close Combat Attacks gain the chosen Attack Attribute. The chosen effect lasts until the end of the phase.%
}
\newcommand{\lordofthedamneddef}{%
\itemrestriction{Sorcerers, Doomlords, and Shrine Priests only.}
Any unit with \irredeemable{} within \distance{18} of the bearer's model may reroll the distance it moves in the Movement Phase with \randommovement{}.%
}
\newcommand{\wyrdstonedef}{%
One use only. Must be activated when the bearer's model suffers the first hit in the game. This hit is ignored. If the bearer is hit by several simultaneous attacks, the bearer chooses which attack to ignore.%
}
