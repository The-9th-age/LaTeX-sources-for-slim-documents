% Special Equipment Texts

\newcommand{\burningportentdef}{%
\enchantmentrestriction{\hw{} enchantment.}
Attacks made with this weapon have their Armour Penetration \textbf{set} to 10 and gain \textbf{\flamingattacks{}}, \textbf{\magicalattacks{}}, and \textbf{\multiplewounds{D3}{}}.
}
\newcommand{\symbolofslaughterdef}{%
\enchantmentrestriction{\hw{} and \pw{} enchantment.}
When using this weapon, the wielder gains +2 Attack Value, +2 Agility, and suffers -2 Offensive Skill. Attacks made with this weapon gain \textbf{\magicalattacks{}}.
}

\newcommand{\thriceforgeddef}{%
\itemrestriction{Standard Size models only.}
\enchantmentrestriction{Suit of Armour enchantment.}
The wearer gains +3 Armour.
}
\newcommand{\gladiatorsspiritdef}{%
\enchantmentrestriction{\spikedshield{} enchantment.}
Hits caused by the \spikedshield{} gain +1 Strength and +1 Armour Penetration.
}

\newcommand{\iconoftheinfinitedef}{%
\itemrestriction{0-2 per Army.}
The bearer can cast \spellformat{\hellfire}{\hereditaryspell} as a Bound Spell with Power Level (4/8).
}
\newcommand{\bannerofthevoiddef}{%
\itemrestriction{Chosen, Chosen Knights, and Chosen Lords only.}
The bearer's unit gains \textbf{\aegis{} (5+, against Ranged Attacks)}.
}
\newcommand{\wastelandtorchdef}{%
The bearer's unit gains \textbf{Strider (Ruins)}. After determining Deployment Zones (at the end of step 6 of the Pre-Game Sequence), you may choose a single Building, Field, or Forest Terrain Feature. Chosen Fields and Forests become Ruins, and chosen Buildings become Cliffs.
}

\newcommand{\ledgerofsoulsdef}{%
\itemrestriction{Dominant.}
The bearer gains \textbf{\thereckoning{}} (see Battle\-shrine).
}
\newcommand{\orbofforebodingdef}{%
At the start of the Deployment Phase, immediately before step 1 (Determine who deploys first), you may choose one of the opponent's non-Character units. This unit loses \vanguard{} and \scout{}. The opponent must immediately deploy this unit. This is done outside the normal Deployment procedure and is ignored when determining the first deployed unit and the number of deployed units.
}
\newcommand{\immortalgauntletsdef}{%
At the start of any Round of Combat you may discard a Veil Token from your Veil Token pool. If so, choose either \divineattacks{}, \flamingattacks{}, or \magicalattacks{}. The bearer's Close Combat Attacks gain the chosen Attack Attribute. The chosen effect lasts until the end of the phase.
}
\newcommand{\lordofthedamneddef}{%
\itemrestriction{Sorcerers only.}
Any unit with \irredeemable{} within \distance{18} of the bearer may reroll the distance it moves in the Movement Phase with \randommovement{}.
}
\newcommand{\wyrdstonedef}{%
One use only. Must be activated when the bearer's model is hit by an attack with at least Strength 5 or at least Armour Penetration 5. The bearer's model gains \textbf{\aegis{4+}}.\newline
If the bearer's model is Standard or Large, this effect is lost at end of the Player Turn.\newline
If the bearer's model is Gigantic, this effect is lost after rolling the \aegis{} Save once (if the bearer is hit by several simultaneous attacks, the owner chooses which attack to use the \aegis{} Save against).
}
