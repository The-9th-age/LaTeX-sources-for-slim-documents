\newcommand{\shapeshifterrestriction}{Models on foot only}
\newcommand{\shapeshifterdef}{%
	The model's Advance Rate is \textbf{set} to \distance{6} and its March Rate is \textbf{set} to \distance{18}. It gains +1 Attack Value, +1 Resilience, \textbf{\fear}, \textbf{\hardtarget{} (1)}, \textbf{\swiftstride}, and \textbf{\vanguard}. The model may never join units or be joined by other Characters.%
}
\newcommand{\wildhunterrestriction}{Models mounted on Elven Horse or Great Elk only}
\newcommand{\wildhunterdef}{%
	The bearer's model gains +1 Attack Value, -2 Defensive Skill, \textbf{\battlefocus{}}, \textbf{\devastatingcharge{} (+1 \At{}, \fear{})}, \textbf{\fearless{}}, \textbf{\frenzy{}}, and \textbf{\lighttroops{}}.%
}
\newcommand{\pathfinderrestriction}{Models on foot only. Cannot be taken by the Battle Standard Bearer}
\newcommand{\pathfinderdef}{%
	The model gains \textbf{\masterarcher} and \textbf{\scout}.\newline
	The Shots of a \sylvanlongbow{} wielded by the model are \textbf{set} to 3. If wielded by a Forest Prince, its Shots are \textbf{set} to 4 instead.%
}
\newcommand{\forestguardianrestriction}{Models on foot only}
\newcommand{\forestguardiandef}{%
	The model gains +1 Attack Value and +1 Armour.%
}
\newcommand{\bladedancerrestriction}{Models on foot only}
\newcommand{\bladedancerdef}{%
	The model gains \textbf{\aegis{6}}, \textbf{\dancesofcenyrn} (see \bladedancer{} unit), and \textbf{\fearless}. The bearer's unit gains \textbf{\swiftstride}. The model may only join or be joined by other \bladedancer{} Kindred Characters and units of \bladedancers{}. It cannot use any Shooting Weapons nor benefit from Armour (neither mundane nor enchanted).%
}
