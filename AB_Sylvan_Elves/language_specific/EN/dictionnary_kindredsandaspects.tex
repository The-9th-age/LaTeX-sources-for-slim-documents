\newcommand{\shapeshifterdef}{%
\itemrestriction{Models on foot only.}
The model has its Advance Rate \textbf{set} to \distance{6} and its March Rate \textbf{set} to \distance{18}. It gains +1 Attack Value, +1 Resilience, \textbf{\fear}, \textbf{\hardtarget{} (1)}, \textbf{\swiftstride}, and \textbf{\vanguard}. The model may never join units or be joined by other Characters.%
}
\newcommand{\wildhunterdef}{%
\itemrestriction{Models mounted on Elven Horse or Great Elk only.}
The bearer's model gains +1 Attack Value, -2 Defensive Skill, \textbf{\battlefocus{}}, \textbf{\devastatingcharge{} (+1 \At{}, \fear{})}, \textbf{\fearless{}}, \textbf{\frenzy{}}, and \textbf{\lighttroops{}}.%
}
\newcommand{\pathfinderdef}{%
\itemrestriction{Models on foot only. Cannot be taken by the Battle Standard Bearer. \zerotoXmodelsperarmy{2}.}
The model gains \textbf{\masterarcher} and \textbf{\scout}.\newline
A \sylvanlongbow{} wielded by the model has its Shots \textbf{set} to 3. If wielded by a Forest Prince, its Shots are \textbf{set} to 4 instead.%
}
\newcommand{\forestguardiandef}{%
\itemrestriction{Models on foot only.}
The model gains +1 Attack Value and +1 Armour.%
}
\newcommand{\bladedancerdef}{%
\itemrestriction{Models on foot only.}
The model gains \textbf{\aegis{6}}, \textbf{\dancesofcenyrn} (see \bladedancer{} unit), and \textbf{\fearless}. The bearer's unit gains \textbf{\swiftstride}. The model may only join or be joined by other \bladedancer{} Kindred Characters and units of \bladedancers{}. It cannot use any Shooting Weapons and benefit from Armour (neither mundane nor enchanted).%
}
