% Special Equipment Texts

\newcommand{\khagadaislegacydef}{%
Great Weapon enchantment.\newline
Attacks made with this weapon gain \textbf{Multiple Wounds(D3)} and become \textbf{Magical Attacks}.
}

\newcommand{\viperscursedef}{%
Brace of Ogre Pistols or Ogre Crossbow enchantment.\newline
This weapon has Shots 4, Str 4, and AP 2 (Range is dependent on which weapon is enchanted). This weapon also gains \textbf{Poison Attacks} and \textbf{Magical Attacks} (in case of Brace of Ogre Pistols, both of these also apply for Close Combat Attacks made with it). An enchanted Ogre Crossbow loses Area Attack (1×5). Shooting Attacks made with this weapon \textbf{always} hit on 4+.
}

\newcommand{\heartripperdef}{%
Iron Fist or Paired Weapons enchantment.\newline
Attacks made with this weapon gain \textbf{Lethal Strike}, +1 Armour Penetration, become \textbf{Magical Attacks}, and can never hit on worse than 3+.
}

\newcommand{\ritualbloodletterdef}{%
\itemrestriction{Shamans only.}
Hand Weapon, Paired Weapons, or Iron Fist enchantment.\newline
The wielder gains +1 Offensive Skill and +1 Attack Value while using this weapon. Attacks made with it become \textbf{Magical Attacks}. For each unsaved wound inflicted with this weapon, the owner gains one Veil Token.
}

\newcommand{\bannerofthegyenggetdef}{%
\itemrestriction{Cannot be taken by units that count towards Core}
All model parts in the bearer’s unit must reroll natural to-hit, to-wound, and Armour Save rolls of ‘1’ in the First Round of Combat (this includes Special Attacks).
}

\newcommand{\pennantofthegreatgrassskydef}{%
The bearer’s unit gains \textbf{Swiftstride}.
}

\newcommand{\skullofqenghetdef}{%
Models in the bearer’s unit gain \textbf{Fear} and automatically pass Panic Tests caused by Terror.
}

\newcommand{\wrestlersbeltdef}{%
\itemrestriction{Models on foot only.}
Light Armour enchantment.\newline
The wearer gains +2 Armour and +1 Strength.
}

\newcommand{\mammothhidecloakdef}{%
\itemrestriction{Models on foot only.}
Suits of Armour enchantment.\newline
The wearer gains +1 Armour. Attacks against the wearer can never have a Strength above 5.
}

\newcommand{\karkadansresiliencedef}{%
Suits of Armour enchantment.\newline
The wearer gains +1 Resilience but automatically fails all Special Saves.
}

\newcommand{\yetifursdef}{%
Suits of Armour enchantment.\newline
The wearer gains +1 Armour. Enemy units in base contact with the wearer suffer \minuss{}1 Agility.
}

\newcommand{\lygurstonguedef}{%
Enemy units in base contact with the bearer suffer \minuss{}1 Attack Value.
}

\newcommand{\aurochscharmdef}{%
The bearer gains \textbf{Mountain Hide}.
}

\newcommand{\rampagerschaindef}{%
The bearer gains \textbf{Stomp (D3+1)}, and all models in the bearer’s unit must reroll failed to-wound rolls with Stomp Attacks.
}
